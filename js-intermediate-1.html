반복문 (Loop)

반복문 = 같은 작업을 여러 번 자동으로 수행

일상생활 예시

100번 "안녕하세요"를 출력하려면?

  • ❌ console.log("안녕하세요")를 100번 복사하기
  • 반복문으로 1번만 작성하고 100번 실행시키기
1

for문 - 횟수가 정해진 반복

// 1부터 5까지 출력
for (let i = 1; i <= 5; i++) {
    console.log("현재 숫자:", i);
}
// 결과: 1, 2, 3, 4, 5
2

while문 - 조건이 참인 동안 반복

let count = 0;
while (count < 3) {
    console.log("카운트:", count);
    count++;  // 중요: 조건을 변경해야 무한루프 방지
}

배열 (Array)

배열 = 여러 개의 데이터를 하나의 변수에 저장

// 배열 만들기
let fruits = ["사과", "바나나", "오렌지"];
let numbers = [1, 2, 3, 4, 5];
let mixed = ["텍스트", 123, true];

// 배열 사용하기
console.log(fruits[0]);  // "사과" (첫 번째 요소)
console.log(fruits.length);  // 3 (배열의 길이)

// 배열에 요소 추가
fruits.push("포도");
console.log(fruits);  // ["사과", "바나나", "오렌지", "포도"]

실습: 반복문과 배열 활용

실습 1: 구구단 만들기

// 5단 구구단
for (let i = 1; i <= 9; i++) {
    let result = 5 * i;
    console.log(`5 × ${i} = ${result}`);
}

실습 2: 학생 명단 관리

let students = ["김철수", "이영희", "박민수", "최지영"];

console.log("우리 반 학생 명단:");
for (let i = 0; i < students.length; i++) {
    console.log(`${i + 1}번: ${students[i]}`);
}
