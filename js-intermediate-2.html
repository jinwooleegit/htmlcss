JavaScript 중급편 2부

함수와 객체로 체계적인 프로그래밍을 배워보세요

함수 (Function)

함수 = 재사용 가능한 코드 블록

// 함수 정의
function greet(name) {
    return `안녕하세요, ${name}님!`;
}

// 함수 호출
let message = greet("홍길동");
console.log(message);  // 안녕하세요, 홍길동님!

객체 (Object)

객체 = 관련된 데이터와 기능을 하나로 묶은 것

// 객체 생성
let person = {
    name: "김철수",
    age: 25,
    job: "개발자",
    introduce: function() {
        return `안녕하세요! 저는 ${this.name}이고, ${this.age}살입니다.`;
    }
};

// 객체 사용
console.log(person.name);  // 김철수
console.log(person.introduce());  // 안녕하세요! 저는 김철수이고, 25살입니다.
