JavaScript 중급편 3부

DOM 조작과 이벤트로 동적 웹페이지 만들기

DOM 조작

DOM = 웹페이지의 HTML 요소를 JavaScript로 제어

// HTML 요소 선택
let element = document.getElementById("myElement");
let buttons = document.getElementsByTagName("button");
let items = document.querySelectorAll(".item");

// 내용 변경
element.textContent = "새로운 텍스트";
element.innerHTML = "강조된 텍스트";

// 스타일 변경
element.style.color = "red";
element.style.fontSize = "20px";

이벤트 처리

// 버튼 클릭 이벤트
document.getElementById("myButton").addEventListener("click", function() {
    alert("버튼이 클릭되었습니다!");
});

// 입력 필드 이벤트
document.getElementById("nameInput").addEventListener("input", function(event) {
    console.log("입력된 값:", event.target.value);
});
