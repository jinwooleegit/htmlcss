⚡ JavaScript란 무엇인가요?

JavaScript는 웹 페이지에 생명을 불어넣는 프로그래밍 언어입니다. HTML이 구조를, CSS가 디자인을 담당한다면, JavaScript는 동적인 기능과 상호작용을 만들어냅니다.

이 텍스트는 JavaScript로 변경될 수 있습니다!

위 데모의 JavaScript 코드:

function changeText() {
    document.getElementById('demo-paragraph').textContent = 
        'JavaScript가 텍스트를 바꿨습니다! 🎉';
}

function changeColor() {
    const colors = ['#ff6b6b', '#4ecdc4', '#45b7d1', '#96ceb4', '#feca57'];
    const randomColor = colors[Math.floor(Math.random() * colors.length)];
    document.getElementById('demo-paragraph').style.color = randomColor;
}

function addElement() {
    const container = document.getElementById('demo-container');
    const newElement = document.createElement('div');
    newElement.textContent = '새로운 요소가 추가되었습니다!';
    newElement.style.background = '#e8f5e8';
    newElement.style.padding = '10px';
    newElement.style.margin = '5px 0';
    newElement.style.borderRadius = '5px';
    container.appendChild(newElement);
}

함수 (Functions)

재사용 가능한 코드 블록을 만드는 방법입니다.

기본 함수

function greet(name) {
    return "안녕하세요, " + name + "님!";
}

// 함수 호출
let message = greet("홍길동");
console.log(message); // "안녕하세요, 홍길동님!"

화살표 함수

const add = (a, b) => {
    return a + b;
};

// 간단한 형태
const multiply = (a, b) => a * b;

console.log(add(5, 3));      // 8
console.log(multiply(4, 2)); // 8

DOM 조작

JavaScript로 HTML 요소를 선택하고 조작하는 방법을 배워보세요.

요소 선택하기

// ID로 선택
let element = document.getElementById('my-id');

// 클래스로 선택
let elements = document.getElementsByClassName('my-class');

// CSS 선택자로 선택
let element = document.querySelector('.my-class');
let elements = document.querySelectorAll('p');

내용 변경하기

// 텍스트 내용 변경
element.textContent = '새로운 텍스트';

// HTML 내용 변경
element.innerHTML = '<strong>굵은 텍스트</strong>';

// 스타일 변경
element.style.color = 'red';
element.style.fontSize = '20px';
