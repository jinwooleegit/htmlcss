연산자 (Operators)

연산자 = 계산과 비교를 위한 기호

쉬운 비유로 이해하기

연산자는 수학 시간에 사용하는 기호와 같아요!

  • +, -, ×, ÷ = 산술 연산자
  • =, ≠, >, < = 비교 연산자
  • 그리고(AND), 또는(OR) = 논리 연산자

JavaScript의 주요 연산자들

➕

산술 연산자

계산
연산자 의미 예시 결과
+ 더하기 5 + 3 8
- 빼기 5 - 3 2
* 곱하기 5 * 3 15
/ 나누기 15 / 3 5
% 나머지 17 % 5 2
++ 1 증가 let x=5; x++ 6
-- 1 감소 let x=5; x-- 4 
// 산술 연산자 예시
let a = 10;
let b = 3;

console.log(a + b);  // 13 (더하기)
console.log(a - b);  // 7  (빼기)
console.log(a * b);  // 30 (곱하기)
console.log(a / b);  // 3.333... (나누기)
console.log(a % b);  // 1  (나머지)

let count = 5;
count++;  // count = count + 1
console.log(count);  // 6
⚖️

비교 연산자

비교
연산자 의미 예시 결과
== 같다 5 == "5" true
=== 정확히 같다 5 === "5" false
!= 다르다 5 != 3 true
!== 정확히 다르다 5 !== "5" true
> 크다 5 > 3 true
< 작다 5 < 3 false
>= 크거나 같다 5 >= 5 true
<= 작거나 같다 3 <= 5 true 
// 비교 연산자 예시
let age = 20;
let name = "홍길동";

console.log(age > 18);      // true (20은 18보다 큰가?)
console.log(age == 20);     // true (나이가 20인가?)
console.log(age === "20");  // false (숫자 20과 문자 "20"은 다름)
console.log(name == "홍길동"); // true (이름이 홍길동인가?)
console.log(age >= 20);     // true (나이가 20 이상인가?)
⚠️ 중요한 차이점
  • ==: 값만 비교 (타입 변환 허용)
  • ===: 값과 타입을 모두 비교 (권장)
🧠

논리 연산자

논리
연산자 의미 설명 예시
&& 그리고 (AND) 모두 true일 때 true true && true → true
|| 또는 (OR) 하나라도 true면 true true || false → true
! 아니다 (NOT) true를 false로, false를 true로 !true → false 
// 논리 연산자 예시
let age = 20;
let hasLicense = true;
let hasExperience = false;

// && (그리고): 모든 조건이 true여야 함
console.log(age >= 18 && hasLicense);  // true (성인이고 면허 있음)
console.log(hasLicense && hasExperience);  // false (면허는 있지만 경험 없음)

// || (또는): 하나라도 true면 됨
console.log(hasLicense || hasExperience);  // true (면허라도 있음)
console.log(age < 18 || !hasLicense);      // false (미성년자가 아니고 면허 있음)

// ! (아니다): 반대로 바꿈
console.log(!hasExperience);  // true (경험이 없다)
console.log(!(age >= 18));    // false (성인이 아니다가 아님)

조건문 (if문)

조건문 = 상황에 따라 다른 행동하기

일상생활 예시

만약(if) 비가 온다면 → 우산을 가져간다

그렇지 않다면(else) → 우산을 가져가지 않는다

프로그래밍도 이와 같은 논리로 작동해요!

if문의 기본 구조

if (조건) {
    // 조건이 true일 때 실행되는 코드
}
if: "만약에"라는 의미
(조건): true 또는 false가 나오는 식
{ }: 실행할 코드를 감싸는 중괄호
1

기본 if문

let age = 20;

if (age >= 18) {
    console.log("성인입니다!");
    alert("투표할 수 있습니다.");
}

결과: age가 18 이상이므로 메시지가 출력됩니다.

2

if-else문

let weather = "비";

if (weather === "비") {
    console.log("우산을 가져가세요!");
} else {
    console.log("좋은 날씨네요!");
}

결과: 비가 오므로 "우산을 가져가세요!" 출력

3

if-else if-else문

let score = 85;

if (score >= 90) {
    console.log("A학점입니다!");
} else if (score >= 80) {
    console.log("B학점입니다!");
} else if (score >= 70) {
    console.log("C학점입니다!");
} else {
    console.log("더 열심히 공부하세요!");
}

결과: 85점이므로 "B학점입니다!" 출력

실습: 조건문 활용하기

실습 1: 나이 확인 프로그램

1

나이별 메시지 출력하기

// 나이 입력
let userAge = 17;  // 여기서 나이를 바꿔가며 테스트해보세요!

// 조건문으로 나이 확인
if (userAge >= 65) {
    console.log("경로우대 대상입니다!");
    console.log("할인 혜택을 받으실 수 있습니다.");
} else if (userAge >= 18) {
    console.log("성인입니다!");
    console.log("모든 서비스를 이용하실 수 있습니다.");
} else if (userAge >= 13) {
    console.log("청소년입니다!");
    console.log("일부 서비스에 제한이 있을 수 있습니다.");
} else {
    console.log("어린이입니다!");
    console.log("보호자 동의가 필요합니다.");
}

userAge 값을 바꿔가며 다른 결과를 확인해보세요!

결과

17살이므로 "청소년입니다!" 메시지가 출력됩니다.

🎯 나이에 따라 다른 메시지가 나타나는 것을 확인하세요!

실습 2: 계산기 프로그램

1

연산자에 따른 계산

// 계산할 숫자와 연산자
let num1 = 10;
let num2 = 5;
let operator = "+";  // +, -, *, / 중 하나

let result;

// 연산자에 따라 다른 계산
if (operator === "+") {
    result = num1 + num2;
    console.log(`${num1} + ${num2} = ${result}`);
} else if (operator === "-") {
    result = num1 - num2;
    console.log(`${num1} - ${num2} = ${result}`);
} else if (operator === "*") {
    result = num1 * num2;
    console.log(`${num1} × ${num2} = ${result}`);
} else if (operator === "/") {
    if (num2 !== 0) {
        result = num1 / num2;
        console.log(`${num1} ÷ ${num2} = ${result}`);
    } else {
        console.log("0으로 나눌 수 없습니다!");
    }
} else {
    console.log("올바른 연산자를 입력해주세요! (+, -, *, /)");
}

operator 값을 "+", "-", "*", "/" 로 바꿔가며 테스트해보세요!

결과

"+" 연산자이므로 "10 + 5 = 15" 가 출력됩니다.

🧮 간단한 계산기를 만들었습니다!

실습 3: 로그인 확인 시스템

1

아이디와 비밀번호 확인

// 저장된 계정 정보
const correctId = "admin";
const correctPassword = "1234";

// 사용자 입력 (실제로는 입력창에서 받아옴)
let userId = "admin";
let userPassword = "1234";

// 로그인 확인
if (userId === correctId && userPassword === correctPassword) {
    console.log("✅ 로그인 성공!");
    console.log("환영합니다, " + userId + "님!");
} else if (userId !== correctId) {
    console.log("❌ 아이디가 올바르지 않습니다.");
} else if (userPassword !== correctPassword) {
    console.log("❌ 비밀번호가 올바르지 않습니다.");
} else {
    console.log("❌ 로그인 정보를 확인해주세요.");
}

// 추가 보안 확인
if (userId === correctId && userPassword === correctPassword) {
    let isVip = true;
    let loginCount = 50;
    
    if (isVip && loginCount > 30) {
        console.log("🌟 VIP 사용자님, 특별 혜택이 준비되어 있습니다!");
    } else if (isVip) {
        console.log("🎯 VIP 사용자로 인증되었습니다.");
    } else if (loginCount > 10) {
        console.log("👍 단골 사용자이시네요!");
    }
}

userId와 userPassword를 바꿔가며 다른 경우를 테스트해보세요!

결과

아이디와 비밀번호가 모두 일치하므로 로그인 성공 메시지가 출력됩니다.

자주하는 실수와 해결법

❌ 실수 1

= 와 == 혼동하기

❌ 잘못된 코드
let age = 20;
if (age = 18) {  // 할당 연산자 사용 (잘못됨)
    console.log("18살입니다");
}
✅ 올바른 코드
let age = 20;
if (age == 18) {  // 비교 연산자 사용 (올바름)
    console.log("18살입니다");
}

해결법: =는 값 할당, ==는 값 비교, ===는 값과 타입 비교

❌ 실수 2

중괄호 빼먹기

❌ 잘못된 코드
if (age >= 18)
    console.log("성인입니다");
    console.log("투표할 수 있습니다");  // 항상 실행됨!
✅ 올바른 코드
if (age >= 18) {
    console.log("성인입니다");
    console.log("투표할 수 있습니다");  // 조건에 따라 실행
}

해결법: 여러 줄의 코드는 반드시 중괄호 { }로 묶어주세요.

❌ 실수 3

논리 연산자 잘못 사용하기

❌ 잘못된 코드
let age = 25;
let hasLicense = false;

// "성인이거나 면허가 있으면"이라고 생각했지만...
if (age >= 18 || hasLicense) {
    console.log("운전할 수 있습니다");  // 잘못된 논리!
}
✅ 올바른 코드
let age = 25;
let hasLicense = false;

// "성인이고 면허도 있어야" 운전 가능
if (age >= 18 && hasLicense) {
    console.log("운전할 수 있습니다");
} else {
    console.log("운전할 수 없습니다");
}

해결법: &&는 "그리고"(모든 조건), ||는 "또는"(하나만 만족)
